int main() {
    int i, j, isPrime, sum = 0;
    for (i = 2; i <= 100; i++) {
        isPrime = 1;    
        for (j = 2; j <= i / 2; j++) {
            if (i % j == 0) {
                isPrime = 0; 
                break;
            }
        }
        if (isPrime) {  
            sum += i;
        }
    }
    printf("Sum of all prime numbers between 1 and 100 is: %d\n", sum);
    return 0;
}
