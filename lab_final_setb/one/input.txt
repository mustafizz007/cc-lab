#include <stdio.h>

int factorial(int n) {
    int f = 1;
    for (int i = 2; i <= n; i++) f *= i;
    return f;
}

int main() {
    int choice, a, b;
    char *menu[] = {"Add", "Subtract", "Factorial"};

    for (int i = 0; i < 3; i++)
        printf("%d. %s\n", i + 1, menu[i]);

    printf("Enter choice: ");
    scanf("%d", &choice);

    if (choice >= 1 && choice <= 3) {
        for (int i = choice; i <= choice; i++) {
            if (i == 1) {
                printf("Enter two numbers: ");
                scanf("%d %d", &a, &b);
                printf("Sum = %d\n", a + b);
            }
            else if (i == 2) {
                printf("Enter two numbers: ");
                scanf("%d %d", &a, &b);
                printf("Difference = %d\n", a - b);
            }
            else {
                printf("Enter number: ");
                scanf("%d", &a);
                printf(a < 0 ? "Factorial not defined for negative numbers.\n"
                              : "Factorial = %d\n", factorial(a));
            }
        }
    } else {
        printf("Invalid choice!\n");
    }
    return 0;
}
